02. Create Data context

Create context with its DbSets
Change name of ConnectionString Name if it is necesary

public class PlutoContext : DbContext
{
	public PlutoContext()
	: base("name=DefaultConnection")
    {}

	DbSet<Course> Courses { get; set; }

	DbSet<Author> Authors { get; set; }

	DbSet<Tag> Tags { get; set; }
}

  <connectionStrings>
    <add name="DefaultConnection" connectionString="data source=FCXLFFOREROG\SqlServer2012; initial catalog=PlutoCodeFirst; integrated security=SSPI" providerName="System.Data.SqlClient" />
  </connectionStrings>
